//version=5

study("Inverse Fisher Transform on STOCHASTIC", shorttitle="IFTSTOCH by Oğuz")
stochlength=input(21, "STOCH Length")
wmalength=input(9, title="Smoothing length")
v1=0.1*(stoch(close, high, low, stochlength)-50)
v2=wma(v1, wmalength)
INV=(exp(2*v2)-1)/(exp(2*v2)+1)
plot(INV, color=blue, linewidth=2)
hline(0.5, color=red)
hline(-0.5, color=green)
hline(0.8, color=blue)
hline(-0.8, color=blue)
